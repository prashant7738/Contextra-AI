CREATE DATABASE contextra_db;
CREATE USER contextra_user WITH PASSWORD 'strong_password_here';
ALTER ROLE contextra_user SET client_encoding TO 'utf8';
ALTER ROLE contextra_user SET default_transaction_isolation TO 'read committed';
ALTER ROLE contextra_user SET timezone TO 'UTC';
GRANT ALL PRIVILEGES ON DATABASE contextra_db TO contextra_user;

GRANT USAGE, CREATE ON SCHEMA public TO contextra_user;
ALTER SCHEMA public OWNER TO contextra_user;
ALTER DATABASE contextra_db OWNER TO contextra_user;
